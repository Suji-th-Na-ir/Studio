using UnityEngine;
using Newtonsoft.Json;

namespace Terra.Studio
{
    [Author("Terra.Studio.REPLACEME")]
    public class REPLACEMEAuthor : BaseAuthor
    {
        public override void Generate(object data)
        {
            var tuple = ((int id, string type, string compData, GameObject obj))data;
            var compData = JsonConvert.DeserializeObject<REPLACEMEComponent>(tuple.compData);
            var ecsWorld = RuntimeOp.Resolve<RuntimeSystem>().World;
            var compPool = ecsWorld.GetPool<REPLACEMEComponent>();
            compPool.Add(tuple.id);
            ref var compRef = ref compPool.Get(tuple.id);
        }
    }
}
